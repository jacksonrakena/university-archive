earnings_long_to2020 <- filter(earnings_long, Date >= '2006-01-01')

find_maximum_earnings_by_industry <- function (data) {
  return(data %>% 
    group_by(name) %>% 
    summarise(max = max(Average.Weekly.Earnings)))
}
